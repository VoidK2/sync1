#include <bits/stdc++.h>
#define eps 1e-8
#define INF 0x3f3f3f3f
#define MOD 1e9+7
#define PI acos(-1)
#define lson l,m,rt<<1
#define rson m+1,r,rt<<1|1
using namespace std;
typedef long long ll;
typedef unsigned long long ull;
const int maxn = 1e5 + 5;

int n, q;
int MIN[maxn << 2];
int lazy[maxn << 2];
int ans[maxn << 2];
int b[maxn << 2];


void pushup(int rt) {
    MIN[rt] = min(MIN[rt << 1], MIN[rt << 1 | 1]);
    ans[rt] = ans[rt << 1] + ans[rt << 1 | 1];
}

void pushdown(int rt) {
    if(lazy[rt]) {
        lazy[rt << 1] += lazy[rt];
        lazy[rt << 1 | 1] += lazy[rt];
        MIN[rt << 1] -= lazy[rt];
        MIN[rt << 1 | 1] -= lazy[rt];
        lazy[rt] = 0;
    }
}
//建树
void build(int l, int r, int rt) {
    lazy[rt] = 0;
    ans[rt] = 0;
    if(l == r) {
        scanf("%d", &b[rt]);
        MIN[rt] = b[rt];
        return ;
    }
    int m = (l + r) >> 1;
    build(lson);
    build(rson);
    pushup(rt);
}

void updata(int L, int R, int l, int r, int rt) {
    if(L <= l && R >= r) {
        MIN[rt]--;
        if(MIN[rt]) {
            lazy[rt]++;
            return;
        } else {
            if(l == r) {
                ans[rt]++;
                MIN[rt] = b[rt];
                return;
            }

        }
    }
    pushdown(rt);
    int m = (l + r) >> 1;
    if(L <= m) updata(L, R, lson);
    if(R > m) updata(L, R, rson);
    pushup(rt);
}
//查询
int query(int L, int R, int l, int r, int rt) {
    if(L <= l && R >= r) {
        return ans[rt];
    }
    pushdown(rt);
    int m = (l + r) >> 1;
    int sum = 0;
    if(m >= L) sum += query(L, R, lson);
    if(m < R) sum += query(L, R, rson);
    return sum;
}
int main() {
    while(~scanf("%d%d", &n, &q)) {
        build(1, n, 1);
        for(int i = 1; i <= q; i++) {
            char op[10];
            int a, b;
            scanf("%s%d%d", op, &a, &b);
            if(op[0] == 'q') {
                printf("%d\n", query(a, b, 1, n, 1));
            } else {
                updata(a, b, 1, n, 1);
            }
        }
    }
    return 0;
}
